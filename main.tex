	\chapter{Introduction}
    \label{chap:introduction}
    \begin{comment}
	Grouped count data arise naturally in many applied settings. In consumer behaviour research, households make repeated purchases across a large number of product categories; in agriculture, field plots are observed repeatedly across seasons with counts recorded for different seed varieties. In both cases, the data share a common structure. Observations are grouped, counts are recorded across $Q$ categories and within-group dependence is an intrinsic feature of the data.
	
	The multinomial distribution provides a natural model for such data, but direct estimation is computationally prohibitive when $Q$ is large. Taddy (2015) addressed this through the Distributed Multinomial Regression (DMR) framework, which exploits the exact Poisson–Multinomial factorisation
	
	\begin{equation}
		\prod_{q=1}^{Q}\mathcal{P}(z_{q};\mu_{q})=\mathcal{P}(S;\sum_{q}\mu_{q})\cdot \mathcal{M}(z;p,S)
	\end{equation}
	to decompose the multinomial regression into $Q$ independent Poisson regressions that can be solved in parallel. Lee, Green and Ryan (2017) extended DMR to grouped data by introducing group-level random effects $\lambda_{j}$, accommodating within-group dependence while retaining the Poisson surrogate structure. However, their model assumes $\lambda_{jq}$ to be independent across categories $q$, which rules out the cross-category correlation that is pervasive in high-dimensional settings. 
    \end{comment}

\section{Likelihood Decomposition}
    Assume $\bm{\gamma}\in \mathbb{R}^{R}$ is the coefficient vector that determines the distribution of $n_{ij}$ with the covariate vector $\bm{x}_{ij}$. The observed data $\bm{Y}$ can be decomposed into two generated models separately:
    \begin{itemize}
    \item Volumn Decision
        \begin{equation}
            n_{ij} \sim \operatorname{Poisson}\!\bigl(\exp(\bm{x}_{ij}^\top\bm{\gamma})\bigr),
        \label{eq:volume}
    \end{equation}
    \item Composition Decision
    \begin{equation}
        \bm{y}_{ij} \mid n_{ij}, \bm{p}_{ij} \sim \operatorname{Multinomial}\bigl(n_{ij}, \bm{p}_{ij}\bigr).
        \label{eg:composition}
    \end{equation}
    \end{itemize}

    Suppose $\bm{\Phi}$ be the collection of all parameters governing the purchase composition $\bm{p}_{ij}$, excluding the covariate vector $\bm{x}_{ij}$ and the coefficient vector $\bm{\gamma}$. For an observation $\bm{y}_{ij}$, the likelihood function of the original data $\bm{y}_{ij}$ is $p\bigl(\bm{y}_{ij}\mid \bm{x}_{ij}, \bm{\gamma},\bm{\Phi}\bigr)$. Then,
    \begin{equation}
        p\bigl(\bm{y}_{ij}\mid \bm{x}_{ij}, \bm{\gamma},\bm{\Phi}\bigr) = \sum_{n=0}^{\infty}  p\bigl(\bm{y}_{ij}, n_{ij}=n \mid \bm{x}_{ij}, \bm{\gamma},\bm{\Phi}\bigr).
        \label{eq:totalprobability}
    \end{equation}
    Since $n_{ij}$ is fixed for an observation, only one term is not equal to 0. Therefore, 
    \begin{align*}
     \log p\bigl(\bm{y}_{ij}\mid \bm{x}_{ij}, \bm{\gamma},\bm{\Phi}\bigr) &=   \log p\bigl(\bm{y}_{ij}, n_{ij} \mid \bm{x}_{ij}, \bm{\gamma},\bm{\Phi}\bigr)\\
     &= \log p\bigl(\bm{y}_{ij}\mid n_{ij}, \bm{\Phi}\bigr)+\log  p\bigl( n_{ij} \mid \bm{x}_{ij}, \bm{\gamma}\bigr)\\
     &= \underbrace{n_{ij}\cdot \bm{x}_{ij}^{\top} \bm{\gamma}-\exp ( \bm{x}_{ij}^{\top} \bm{\gamma})}_{\ell_{\mathrm{vol}(\gamma)}}-\underbrace{\sum_{q} \log (y_{ijq}!)}_{\mathrm{constant}}+\underbrace{\sum_{q}y_{ijq}\log (p_{ijq})}_{\ell_{comp}(p_{ijq})}.
    \end{align*}
    The logarithmic likelihood separates into $\ell_{vol}(\bm{\gamma})$, which depends only on $\bm{\gamma}$,  and $\ell_{comp}(\Phi)$, which depends only on $\bm{\Phi}$. The constant term does not depend on any parameter. 
    \begin{comment}
    \begin{equation}
        p(\boldsymbol{y}_{ij}) \;=\; p(\boldsymbol{y}_{ij}\mid n_{ij})\cdot p(n_{ij}),
        \label{eq:chain}
    \end{equation}
  \end{comment}
    Hence, the data set $\bm{Y}$ can be decomposed into two disjoint models under the assumption of two decisions, and the further analysis of latent factors can be structured on the composition decision only, which is not related to the total count number.
    \begin{remark}[Parameter separation]
\label{rem:separation}
The cross-partial derivative $\frac{\partial^2\ell}{\partial\bm{\gamma}\,\partial\bm{\Phi}}=\bm{0}$ everywhere. Consequently, the maximum likelihood estimator of $\bm{\Phi}$ is obtained by maximizing $\ell_{\mathrm{comp}}(\bm{\Phi})$ alone. The composition parameters are estimated entirely from the relative allocation across count categories and free of confounding by volume heterogeneity. This is a principal advantage over Poisson-lognormal models such as \cite{chiquet2018}, in which volume and composition are entangled in a single Poisson likelihood.
\end{remark}
    
Now, a pure multinomial distribution has been obtained from the original data set $\bm{Y}$, and inference of the latent factor structure can proceed by maximizing:
\begin{equation}
    \ell_{comp}(\Phi)=\sum_{i,j}\sum_{q} y_{ijq}\log (p_{ijq}).
    \label{eq:complink}
\end{equation}
The log-likelihood of the multinomial model with unknown probability vector $\bm{p}_{ij}$ and the remaining task is to specify a flexible yet parsimonious structure for $\bm{p}_{ij}$ that addresses two objectives:
\begin{description}
\item[Objective~1 (High-dimensional $Q$).] When $Q$ is large,  estimating $\bm{p}_{ij}$ requires $O(m\,n\,Q)$ parameters, whose cost is expensive and unidentifiable without structure. We seek a low-rank decomposition that compresses the $Q$-dimensional probability vector into a $K$-dimensional latent space, $K\ll Q$.
\item[Objective~2 (Cross-category Co-variation).] Categories do not change independently across $j$ groups, so we want to design a model that captures the group-level effects between categories and do the inferential analyze.
\end{description}
\section{Parameter Estimation Modeling}
 Following the framework of \cite{lee2017poissontrickextensionsfitting}, we introduce two sets of positive latent quantities:
 \begin{itemize}
 \item $\zeta_{iq}$: an individual-level fixed effect for individual $i$ on category $q$ capturing preference heterogeneity for an observation.
 \item  $\lambda_{jq}$: a group-level random effect for group $j$ and category $q$ capturing the variation happens in the whole group.
\end{itemize}
We define the probability as the normalized product of these two quantities:
\begin{equation}
    p_{ijq}=\dfrac{\lambda_{jq}\,\zeta_{iq}}{\sum_{q'} \lambda_{jq'}\,\zeta_{iq'}}
    \label{eq:probproduct}
\end{equation}
    \begin{lemma}[Poisson-Multinomial Factorisation]\label{PMF}
		Let $Z_{1},\dots ,Z_{Q}$ be independent and $Z_{q}\sim \mathcal{P}(\mu_{q})$ for $q=1,2,\dots ,Q$, where $\mu_{q}>0$. Define $S=\sum_{q=1}^{Q}Z_{q}$ and the probability vector $p=(\frac{\mu_1}{\mu_{+}},\dots ,\frac{\mu_Q}{\mu_{+}})^{\top}$, where $\mu_{+}=\sum_{q=1}^{Q}\mu_{q}$.Then the joint distribution of $(Z_{1},\dots,Z_{Q})$ factorises exactly as:
		\begin{equation}
			p(Z_{1}=z_{1},\dots ,Z_{Q}=z_{Q})=\mathcal{P}(S=s;\mu_{+})\cdot \mathcal{M}(z;p,s)
		\end{equation}
		where $s=\sum_{q=1}^{Q}z_{q}$, $\mathcal{P}(\sim ;\mu_{+})$ denotes the Poisson probability mass function with poisson parameter $\mu_{+}$ and $\mathcal{M}(\sim ;p,s)$ denotes the multinomial probability mass function given total $s$ and probability vector $p$.
	\end{lemma}
	
	\begin{proof}
		Since $Z_{1},\dots ,Z_{Q}$ are independent with $Z_{q}\sim \mathcal{P}(\mu_{q})$, by direct computation:
		\begin{equation}
			\prod_{q=1}^{Q} \frac{e^{-\mu_{q}}\mu_{q}^{z_{q}}}{z_{q}!}=e^{-\mu_{+}}\cdot \frac{\mu_{+}^{s}}{s!}\cdot \frac{s!}{\prod_{q=1}^{Q}z_{q}!} \prod_{q=1}^{Q}(\frac{\mu_{q}}{\mu_{+}})^{z_{q}}= \mathcal{P}(s;\mu_{+})\cdot\mathcal{M}(z;p,s).
		\end{equation}
	\end{proof}
	
	\begin{remark}
		Lemma has an important implication for inference. 
	\end{remark}
	
	\subsection{Conditional Model Given Random Effects}
	To discuss with-in group dependence, we introduce a Q-dimensional vector of group-specific random effects $\lambda_{j}=(\lambda_{j1},\dots ,\lambda_{jQ})^{\top}$, where $\lambda_{jq}>0$ represents the multiplicative deviation of group $j$'s baseline intensity for category $q$ from the population average. Conditional on $\lambda_{j}$, we assume that all counts within group $j$ are independent:
	\begin{equation}\label{conditional}
		Y_{ijq}|\lambda_{j} \sim \mathcal{P}(\delta_{ij}\lambda_{jq}\zeta_{ijq}), i=i,\dots ,n_{j}, q=1,\dots ,Q
	\end{equation}
where the fixed-effect component of the conditional mean is:
	\begin{equation}
		\zeta_{ijq}=\exp(v_{ij}^{\top}\phi_{q})
	\end{equation}
and $\phi_{q}\in \mathbb{R}^{p}$ is a vector of regression coefficients for category $q$. The random effect $\lambda_{jq}$ captures the persistent tendency of group $j$ toward category $q$ that is not explained by the observed covariates $v_{ij}$. Observations within the same group share one same $\lambda_{j}$, which induces within-group dependence even after conditioning on covariates.
	\begin{remark}[Conditional Poisson-Multinomial Equivalence]
		By Lemma \ref{PMF} applied conditionally, model \ref{conditional} implies that:
		\begin{equation}
			Y_{ij}|Y_{ij+}, \lambda_{j}\sim  \mathcal{M}(Y_{ij+},p_{ij}), \text{ where } p_{ijq}=\dfrac{\lambda_{jq}\zeta_{ijq}}{\sum_{q=1}^{Q}\lambda_{jq}\zeta_{ijq}}
		\end{equation}
	\end{remark}
    Given the random effects, we treat $Y_{ij+}$ as random and assume 
		\begin{equation}
			Y_{ij+}|\lambda_{j}\sim \mathcal{P}(\delta_{ij}\sum_{q=1}^{Q}\lambda_{jq}\zeta_{ijq}).
		\end{equation}
independently for each $i$ and $j$. This gives
    \begin{equation}
        Y_{ijq}|\lambda_{jq}\sim \mathcal{P}(\delta_{ij}\lambda_{jq}\zeta_{ijq}),
    \end{equation}
independently for each $j$ and $q$. Now the relationship of likelihood between multinomial and poisson models satisfies 
    \begin{equation}\label{ori_likelihood}
        \sum_{j} \sum_{i} \ell^{\mathcal{P}} \left( \delta_{ij} \sum_{q=1}^{Q} \lambda_{jq} \zeta_{ijq};\, Y_{ij+} \Big| \lambda_j \right) + \sum_{j} \sum_{i} \ell^{\mathcal{M}} \left( \zeta_{ij};\, Y_{ij} \Big| Y_{ij+},\, \lambda_j \right) = \sum_{j} \sum_{i} \sum_{q} \ell^{\mathcal{P}} \left( \delta_{ij} \lambda_{jq} \zeta_{ijq};\, Y_{ijq} \Big| \lambda_{jq} \right).
    \end{equation}
	However, the random effects are unobserved so we consider the marginal distribution obtained by integrating out the random effects. The marginal likelihood of the approximating poisson surrogate model (right hand side of Equation \ref{ori_likelihood}) can be expressed in closed-form if we assume an independent Gamma model for the random effect, i.e. $\lambda_{jq}\sim G(\alpha, \beta)$. Then the marginal likelihood of the multinomial model that we would like to fit (second term on the left hand side of Equation \ref{ori_likelihood}) is given by
    \begin{equation}
        L^{\mathcal{M}} = \prod_{j} \int \cdots \int \prod_{i} \left\{ \frac{y_{ij+}!}{\prod_{q} y_{ijq}!} \prod_{q} \left( \frac{\lambda_{jq} \zeta_{ijq}}{\sum_{q=1}^{Q} \lambda_{jq} \zeta_{ijq}} \right)^{y_{ijq}} \right\} \times \prod_{q} \frac{\beta_{q}^{\alpha_q} \lambda_{jq}^{\alpha_q - 1} e^{-\beta_q \lambda_{jq}}}{\Gamma(\alpha_q)} \, \mathrm{d}\lambda_{j1} \cdots \mathrm{d}\lambda_{jQ}.
    \end{equation}
which does not generally exhibit a closed-form solution regardless of the random effect distribution. On the other hand, the marginal likelihood of the Poisson surrogate model can be expressed in closed-form
    \begin{align}
        L^{\mathcal{P}} &= \prod_{i} \left\{ \int \cdots \int \prod_{j} \prod_{q} \frac{e^{-\delta_{ij} \lambda_{iq} \zeta_{ijq}} (\delta_{ij} \lambda_{iq} \zeta_{ijq})^{y_{ijq}}}{y_{ijq}!} \prod_{q} \frac{\beta_{q}^{\alpha_q} \lambda_{iq}^{\alpha_q - 1} e^{-\beta_q \lambda_{iq}}}{\Gamma(\alpha_q)} \, \mathrm{d}\lambda_{i1} \cdots \mathrm{d}\lambda_{iQ} \right\} \nonumber \\
        &= \prod_{i} \left\{ \prod_{q} \frac{\Gamma(\alpha_q + y_{i+q}) \beta_{q}^{\alpha_q}}{\Gamma(\alpha_q)(\beta_q + \sum_{j} \delta_{ij} \zeta_{ijq})^{\alpha_q + y_{i+q}}} \times \prod_{j} \prod_{q} \frac{(\delta_{ij} \zeta_{ijq})^{y_{ijq}}}{y_{ijq}!} \right\}.
    \end{align}

    \begin{comment} 
        The Poisson-Multinomial equivalence, therefore, is exactly at the conditional level, given $\lambda_{j}$. The multinomial probability vector $p_{ij}$ depends on $\lambda_{j}$ only through the ratios $\frac{\lambda_{jq}}{\sum_{q}\lambda_{jq}}$, so the random effects act as multiplicative modulation of the category probabilities that varies across groups but remains constant within a group across observations. 
    \end{comment}

        There are some disadvantages when we use the Poisson surrogate model directly, and we want to find the low-rank structure of matrix. Hence factor-structured random effect is introduced.

	\subsection{Factor-Structured Random Effects}
	The cross-category correlation is driven by common latent factors rather than being independently determined for each category. We proposes a factor-structured random effects model of $\boldsymbol{\lambda}_{j}$ across groups.  $\lambda_{jq}$ are independent across $q$, so the cross-category covariance of $\boldsymbol{\lambda}_{j}$ is diagonal. This assumption becomes increasingly restrictive as $Q$ grows, since it rules out the systematic co-variation of preferences across categories that is typical in high-dimensional count data. We relax this assumption by modelling the log-transformed random effects through a latent factor model:
	\begin{equation}
		\log \boldsymbol{\lambda}_{j}=\boldsymbol{\mu}+\mathbf{B}\boldsymbol{f}_{j}+\boldsymbol{\epsilon}_{j}
	\end{equation}
	where the components are defined as follows. The vector $\boldsymbol{\mu}=(\mu_{1},\dots ,\mu_{Q})^{\top}\in \mathbb{R}^{Q}$ denotes the population-level baseline log-intensities for each category, the matrix $\mathbf{B}=[\boldsymbol{b}_{1},\dots ,\boldsymbol{b}_{K}]\in \mathbb{R}^{Q\times K}$ is the factor loading matrix with $K\ll Q$ and its entries $\mathbf{B}_{qk}$ denote the sensitivity of category $q$'s random effect to the $k$-th latent factor. The vector $\boldsymbol{f}_{j}\in \mathbb{R}^{K}$ denotes the group-specific latent factor scores assumed as follows, distributed identically and independently:
	\begin{equation}
		\boldsymbol{f}_{j}\sim \mathcal{N}_{K}(0,\mathbf{I}_{K})
	\end{equation}
	The vector $\boldsymbol{\epsilon}_{j}\in \mathbb{R}^{Q}$ denotes the error term capturing category-specific variation in $\boldsymbol{\lambda}_{j}$ that is not attributable to the K common factors:
	\begin{equation}
		\boldsymbol{\epsilon}_{j}\sim \mathcal{N}_{Q}(0,\mathbf{D}), \quad \mathbf{D}=\text{diag}(d_{1}^{2},\dots d_{Q}^{2})
	\end{equation} 
	with $\boldsymbol{f}_{j}\perp \boldsymbol{\epsilon}_{j}$ for all $j$. Then the marginal distribution of $\log \boldsymbol{\lambda}_{j}$ is:
	\begin{equation}
		\log \boldsymbol{\lambda}_{j}\sim \mathcal{N}_{Q}(\boldsymbol{\mu}, \boldsymbol{\Sigma}),\quad \boldsymbol{\Sigma}=\mathbf{B}\mathbf{B}^{\top}+\mathbf{D}
	\end{equation}
    \begin{proof}
Under the latent factor model,
    \[
    \log \boldsymbol{\lambda}_j \mid \mathbf{f}_j
    =
    \boldsymbol{\mu} + \mathbf{B}\mathbf{f}_j + \boldsymbol{\varepsilon}_j,
    \qquad
    \boldsymbol{\varepsilon}_j \sim \mathcal{N}_Q(\mathbf{0}, \mathbf{D}),
    \]
so that
    \[
    \log \boldsymbol{\lambda}_j \mid \mathbf{f}_j
    \sim \mathcal{N}_Q(\boldsymbol{\mu}+\mathbf{B}\mathbf{f}_j, \mathbf{D}).
    \]

By the law of total expectation,
\[
\mathbb{E}(\log \boldsymbol{\lambda}_j)
=
\mathbb{E}\!\left[\mathbb{E}(\log \boldsymbol{\lambda}_j\mid \mathbf{f}_j)\right]
=
\boldsymbol{\mu}+\mathbf{B}\mathbb{E}(\mathbf{f}_j)
=
\boldsymbol{\mu},
\]
where \(\mathbb{E}(\mathbf{f}_j)=\mathbf{0}\).

By the law of total variance,
\[
\operatorname{Var}(\log \boldsymbol{\lambda}_j)
=
\mathbb{E}\!\left[\operatorname{Var}(\log \boldsymbol{\lambda}_j\mid \mathbf{f}_j)\right]
+
\operatorname{Var}\!\left[\mathbb{E}(\log \boldsymbol{\lambda}_j\mid \mathbf{f}_j)\right]
=
\mathbf{D}+\mathbf{B}\mathbf{B}^{\top},
\]
since \(\operatorname{Var}(\mathbf{f}_j)=\mathbf{I}_K\).

Hence the covariance matrix has the factor-analytic form
\[
\mathbf{\Sigma}=\mathbf{D}+\mathbf{B}\mathbf{B}^{\top}.
\]

For two distinct categories \(q\) and \(q'\),
\[
\log \lambda_{jq}-\mu_q=\mathbf{b}_q^T\mathbf{f}_j+\varepsilon_{jq},
\quad
\log \lambda_{jq'}-\mu_{q'}=\mathbf{b}_{q'}^T\mathbf{f}_j+\varepsilon_{jq'}.
\]
Therefore,
\[
\operatorname{Cov}(\log \lambda_{jq},\log \lambda_{jq'})
=
\mathbf{b}_q^T\operatorname{Var}(\mathbf{f}_j)\mathbf{b}_{q'}
=
\mathbf{b}_q^T\mathbf{b}_{q'},
\]
where we use the independence between \(\mathbf{f}_j\) and \(\boldsymbol{\varepsilon}_j\), together with the fact that \(D\) is diagonal.
\end{proof}

    The covariance matrix $\Sigma$ has a factor-analytic structure. The rank-$K$  positive semi-definite matrix $\mathbf{B}\mathbf{B}^{\top}$  captures the cross-category covariation of $\log \lambda_{j}$  driven by $K$  shared latent processes, and the matrix $\mathbf{D}$  captures the residual category-specific variation. The cross-category covariance between two different categories has the following property:
    \begin{equation}
    \operatorname{Cov}(\log \lambda_{jq}, \log \lambda_{jq^{'}})=\mathbf{b}_q^T\mathbf{b}_{q'}
    \end{equation}
which is nonzero whenever categories $q$  and $q^{'}$  share the common factor loading.
\begin{proof}
    By the definition of $\log \lambda_{j}$, we obtain that 
\[\log \lambda_{jq}-\mu_{q}=b^{T}_{q}f_{j}+\epsilon_{jq}\]
\[\log \lambda_{jq^{'}}-\mu_{q^{'}}=b^{T}_{q^{'}}f_{j}+\epsilon_{jq^{'}}\]
    Thus, the covariance is calculated by
\[
\begin{aligned}
\operatorname{Cov}\!\bigl(\log \lambda_{jq}, \log \lambda_{jq'}\bigr)
&= \mathbb{E}\Bigl[\bigl(\mathbf{b}_q^{T}\mathbf{f}_j + \varepsilon_{jq}\bigr)
\bigl(\mathbf{b}_{q'}^{T}\mathbf{f}_j + \varepsilon_{jq'}\bigr)\Bigr] \\

 + \mathbb{E}\!\bigl[\varepsilon_{jq}\varepsilon_{jq'}\bigr] \\
&= \mathbf{b}_q^{\top}\mathbf{b}_{q'} .
\end{aligned}
\]
since we assume  $f_{j}\perp \epsilon_{j}$  for all $j$ . 
\end{proof}

\section{Logistic Normal Multinomial Factor Analyzers for Clustering Microbiome Data \cite{Tu2023}}
\subsection{Introduction}
Logistic normal multinomial mixture model (LMM-MM) was proposed for clustering microbiome data and a family of logistic normal multinomial factor analyzers (LNM-FA) was proposed by incorporating a factor analyzer structure in LNM-MM for high-dimensional dataset as the number of free parameters in LNM-FA can be greatly reduced by assuming that the number of latent factor is small.
Using Euclidean geometry on such compositional datasets is not appropriate, when the compositional vectors are added or multiplied, the results may end up outside the simplex and Euclidean difference does not reveal the true difference in relative abundance. Therefore, it is important to take into account the compositional nature of the data. 
A natural choice for modelling the compositional data is a multinomial model with probability vector $\boldsymbol{p}=(p_{1},\cdots,p_{Q})^{\top}$.

\subsection{Methodology}
\subsubsection{Additive Logistic Normal Multinomial Model}
For the counting observation $\boldsymbol{y}\in \mathbb{R}^{Q}$, the multinomial distribution conditional on the probability vector $\boldsymbol{p}=(p_{1},\cdots ,p_{Q})^{\top}$ and the total sample count $n$ is,
\begin{equation}
f(\boldsymbol{y}\mid \boldsymbol{p},n)=\dfrac{n!}{\prod_{q=1}^{Q}y_{q}!}\prod_{q=1}^{Q}p_{q}^{y_{q}} \propto  \prod_{q=1}^{Q}p_{q}^{y_{q}}
\end{equation}
where $\frac{n!}{\prod_{q=1}^{Q}y_{q}!}$ depends only on the data and $\prod_{q=1}^{Q}p_{q}^{y_{q}}$ is the product of probability. 

\subsubsection{The Map from Simplex to $\mathbb{R}^{Q}$}
For each observation, the probability vector $\boldsymbol{p}$ is restricted by the constraint $\sum_{q=1}^{Q}p_{q}=1$, $p_{q}>0$, which does not allow us to use any other distribution in this space. Therefore, a map is necessary from restricted space to open real space $\mathbb{R}^{Q-1}$. Thus, an additive log-ratio (ALR) transformation is utilized to map $\boldsymbol{p}$ from the restricted simplex $\mathbb{S}^{Q-1}$ to a (Q-1)-dimensional open real space $\mathbb{R}^{Q-1}$ such that
\begin{equation}
\boldsymbol{W}=\phi(\boldsymbol{p})=\Bigl[\log \Bigl(\dfrac{p_{1}}{p_{Q}}\Bigr),\cdots , \log \Bigl(\dfrac{p_{Q-1}}{p_{Q}}\Bigr)\Bigr]^{\top},
\end{equation}
where $p_{Q}$ is used as a reference and a multivariate Gaussian distribution is imposed with mean $\boldsymbol{\mu}$ and covariance $\boldsymbol{\Sigma}$ on $\boldsymbol{W}$. Here $\phi: (0,1)^{Q-1} \to \mathbb{R}^{Q-1}$ is a one-to-one function. Therefore,
\begin{equation}
\boldsymbol{p}=\phi^{-1}(\boldsymbol{W})=\Bigl[ \dfrac{\exp{(W_{1})}}{\sum_{q=1}^{Q-1}\exp{(W_{q})}+1},\cdots , \dfrac{\exp{(W_{q})}}{\sum_{q=1}^{Q-1}\exp{(W_{q})}+1}, \dfrac{1}{\sum_{q=1}^{Q-1}\exp{(W_{q})}+1}\Bigr]^{\top}.
\end{equation}
Thus, the conditional probability function of $\boldsymbol{Y}\mid \boldsymbol{W}$ is 
\begin{equation}
f(\boldsymbol{y}\mid \boldsymbol{w}) \propto \prod_{q=1}^{Q-1}\Bigl(\dfrac{\exp{(w_{q})}}{\sum_{q=1}^{Q-1}\exp{(w_{q})}+1}\Bigr)^{y_{q}} \Bigl(\dfrac{1}{\sum_{q=1}^{Q-1}\exp{(w_{q})}+1}\Bigr)^{y_{Q}},
\end{equation}
Here we have two layers, one is the parameter layer assuming $\boldsymbol{W}\sim \mathcal{N}(\boldsymbol{\mu}, \boldsymbol{\Sigma})$, and the observed layer is  $\boldsymbol{Y}\mid \boldsymbol{W}\sim \mathcal{M}(\phi^{-1}(\boldsymbol{W}))$. The counts of observation $\boldsymbol{Y}$ do not depend on the parameters $\boldsymbol{\mu}$ and $\boldsymbol{\Sigma}$ directly, but they characterize the distribution of $\boldsymbol{W}$. When we marginalize over $\boldsymbol{W}$ by integrating $\boldsymbol{W}$ from the joint distribution of $(\boldsymbol{Y},\boldsymbol{W})$, the parameters are propagated into the marginal distribution of $\boldsymbol{Y}$ through the following identity,   
\begin{align}
f(\boldsymbol{y}\mid \boldsymbol{\mu}, \boldsymbol{\Sigma})&=\int_{\mathbb{R}^{Q-1}} f(\boldsymbol{y}\mid \boldsymbol{w}) f(\boldsymbol{w}\mid \boldsymbol{\mu}, \boldsymbol{\Sigma})\,d\boldsymbol{w}\\
&\propto \int_{\mathbb{R^{Q-1}}} \prod_{q=1}^{Q} \bigl(\phi^{-1}(\boldsymbol{w}_{q})\bigr)^{y_{q}}  |\boldsymbol{\Sigma}|^{-\frac{1}{2}}\exp\Bigl[-\dfrac{1}{2}(\boldsymbol{y}-\boldsymbol{\mu})^{\top}\boldsymbol{\Sigma}^{-1}(\boldsymbol{y}-\boldsymbol{\mu})\Bigr]\,d\boldsymbol{w}.
\end{align}
This marginal probability function of $\boldsymbol{Y}$ involves multiple integrals and cannot be further simplified.
\subsubsection{Mixtures of the Logistic Normal Multinomial Models (LNM-MM)}
A computationally efficient framework for clustering data was proposed for parameter estimation using variational Gaussian approximations. Using an approximating density $q(\boldsymbol{w})$, the marginal log density of $\boldsymbol{Y}$ follows,
\begin{align}
\log f(\boldsymbol{y})
&= \int q(\boldsymbol{w}) \log \frac{q(\boldsymbol{w})}{f(\boldsymbol{w}\mid \boldsymbol{y})}\, d\boldsymbol{w}
 + \int q(\boldsymbol{w}) \log \frac{f(\boldsymbol{y}, \boldsymbol{w})}{q(\boldsymbol{w})}\, d\boldsymbol{w} \\
&= D_{KL}\!\bigl[q(\boldsymbol{w}) \,\|\, f(\boldsymbol{w}\mid \boldsymbol{y})\bigr]
 + F\bigl(q(\boldsymbol{w}), \boldsymbol{y}\bigr).
\end{align}
where $F(q(\boldsymbol{w}),y)$ is known as the evidence lower bound (ELBO, \cite{Blei03042017}). Since the left side of the equation only depends on $\boldsymbol{y}$, which is fixed for each observation, so we maximize ELBO and obtain $q(\boldsymbol{w})$ that is very close to $f(\boldsymbol{w}\mid \boldsymbol{y})$.  In VGA framework, $q(\boldsymbol{w})$ is taken to be Gaussian distribution with mean $\boldsymbol{m}$ and variance $\boldsymbol{V}$. VGA framework reduces the computational needs for an MCMC-based approach for parameter estimation and extends for clustering in high-dimensional setting.
\subsubsection{Mixtures of Logistic Normal Multinomial Factor Analyzers}
A finite mixture of LNM model with $n$-component can be written as,
\begin{equation}
f(\boldsymbol{y}_{ij}\mid \boldsymbol{\vartheta})=\sum_{j=1}^{n}\pi_{j}f(\boldsymbol{y}_{ij}\mid\boldsymbol{\mu}_{j},\boldsymbol{\Sigma}_{j}),
\end{equation}
where $\vartheta$ denotes the parameter space, $\pi_{j}$ denotes the mixing proportion of  the $j$-th population and $f(\boldsymbol{y}_{ij}\mid \boldsymbol{\vartheta})$ denotes the marginal probability mass function of the logistic normal multinomial model of $i$-th observation in $j$-th population. The likelihood of the mixture is the product of each probability,
\begin{equation}
L(\boldsymbol{\vartheta})=\prod_{i=1}^{m}\sum_{j=1}^{n}\pi_{j}f(\boldsymbol{y}_{ij}\mid\boldsymbol{\mu}_{j},\boldsymbol{\Sigma}_{j})
\end{equation}
However, the expression of likelihood is not suitable for further computation because it contains both sum and product. So we define a group membership indicator variable $\boldsymbol{z}_{i}=(z_{i1},\cdots ,z_{iQ})^{\top}$ such that $z_{iq}=1$ when the $i$-th observation belongs to group $j$ and 0 otherwise. In the context of clustering, these group memberships are treated as unobserved or missing data and function $L(\boldsymbol{\vartheta})$ is considered an incomplete-data likelihood function. Then the complete-data likelihood with observed data $\boldsymbol{Y}$ and missing data $\boldsymbol{Z}$ can be written as
\begin{equation}
L(\boldsymbol{\vartheta})=\prod_{i=1}^{m}\prod_{j=1}^{n}\Bigl[\pi_{j}f(\boldsymbol{y}_{ij}\mid\boldsymbol{\mu}_{j},\boldsymbol{\Sigma}_{j})\Bigr]^{z_{ij}}.
\end{equation}
and the complete-data log-likelihood becomes
\begin{equation}
\mathcal{l}(\boldsymbol{\vartheta})=\sum_{i=1}^{m}\sum_{j=1}^{n}z_{ij}\Bigl[\log \pi_{j}+\log f(\boldsymbol{y}_{ij}\mid\boldsymbol{\mu}_{j},\boldsymbol{\Sigma}_{j})\Bigr].
\end{equation}
The factor analyzer structure in the mixtures of LNM models utilize the following structure on $\boldsymbol{W}$ from the $j$-th component:
\begin{equation}
\boldsymbol{W}_{i}=\boldsymbol{\mu}_{j}+\boldsymbol{\mathbf{B}}_{j}\mathbf{f}_{ij}+\boldsymbol{\epsilon}_{ij}
\end{equation}
